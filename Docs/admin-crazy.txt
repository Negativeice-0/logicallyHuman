<?php
require_once "config.php";
requireAdmin(); // This now uses named placeholders internally

$db = getDB();
$error = "";
$success = "";

// Handle post deletion - NAMED PLACEHOLDER REFACTOR: :post_id
if (isset($_POST["delete_post"])) {
    $post_id = $_POST["post_id"];
    try {
        // Delete comments first - NAMED PLACEHOLDER: :post_id
        $stmt = $db->prepare("DELETE FROM comments WHERE post_id = :post_id");
        $stmt->execute([":post_id" => $post_id]);

        // Delete post - NAMED PLACEHOLDER: :post_id
        $stmt = $db->prepare("DELETE FROM posts WHERE id = :post_id");
        $stmt->execute([":post_id" => $post_id]);
        $success = "Post deleted successfully!";
    } catch (PDOException $e) {
        $error = "Error deleting post: " . $e->getMessage();
    }
}

// Handle user management - NAMED PLACEHOLDER REFACTOR: :role, :user_id
if (isset($_POST["update_user_role"])) {
    $user_id = $_POST["user_id"];
    $new_role = $_POST["role"];

    // Prevent self-demotion
    if ($user_id == $_SESSION["user_id"] && $new_role != "admin") {
        $error = "You cannot remove your own admin privileges!";
    } else {
        try {
            $stmt = $db->prepare(
                "UPDATE users SET role = :role WHERE id = :user_id",
            );
            $stmt->execute([":role" => $new_role, ":user_id" => $user_id]);
            $success = "User role updated successfully!";

            // Update session if current user updated themselves
            if ($user_id == $_SESSION["user_id"]) {
                $_SESSION["role"] = $new_role;
            }
        } catch (PDOException $e) {
            $error = "Error updating user: " . $e->getMessage();
        }
    }
}

// Handle user deletion - NAMED PLACEHOLDER REFACTOR: :user_id
if (isset($_POST["delete_user"])) {
    $user_id = $_POST["user_id"];

    // Prevent self-deletion
    if ($user_id == $_SESSION["user_id"]) {
        $error = "You cannot delete your own account!";
    } else {
        try {
            // Start transaction
            $db->beginTransaction();

            // Delete user's comments - NAMED PLACEHOLDER: :user_id
            $stmt = $db->prepare(
                "DELETE FROM comments WHERE user_id = :user_id",
            );
            $stmt->execute([":user_id" => $user_id]);

            // Delete user's posts - NAMED PLACEHOLDER: :user_id
            $stmt = $db->prepare("DELETE FROM posts WHERE user_id = :user_id");
            $stmt->execute([":user_id" => $user_id]);

            // Delete user - NAMED PLACEHOLDER: :user_id
            $stmt = $db->prepare("DELETE FROM users WHERE id = :user_id");
            $stmt->execute([":user_id" => $user_id]);

            $db->commit();
            $success = "User deleted successfully!";
        } catch (PDOException $e) {
            $db->rollBack();
            $error = "Error deleting user: " . $e->getMessage();
        }
    }
}

// Handle comment deletion - NAMED PLACEHOLDER REFACTOR: :comment_id
if (isset($_POST["delete_comment"])) {
    $comment_id = $_POST["comment_id"];
    try {
        $stmt = $db->prepare("DELETE FROM comments WHERE id = :comment_id");
        $stmt->execute([":comment_id" => $comment_id]);
        $success = "Comment deleted successfully!";
    } catch (PDOException $e) {
        $error = "Error deleting comment: " . $e->getMessage();
    }
}

// Get all users (no parameters needed)
$users_stmt = $db->query(
    "SELECT id, username, email, role, created_at FROM users ORDER BY created_at DESC",
);
$users = $users_stmt->fetchAll();

// Get all posts with author info (no parameters needed)
$posts_stmt = $db->query("
    SELECT p.*, u.username as author_name
    FROM posts p
    JOIN users u ON p.user_id = u.id
    ORDER BY p.created_at DESC
");

    SELECT c.*, u.username, p.title as post_title

    JOIN posts p ON c.post_id = p.id
    ORDER BY c.created_at DESC
");
$comments = $comments_stmt->fetchAll();

        // Smooth scrolling for admin sidebar navigation
        document.querySelectorAll('.admin-sidebar a').forEach(link => {
            link.addEventListener('click', function(e) {
                e.preventDefault();
                const targetId = this.getAttribute('href').substring(1);
                const targetSection = document.getElementById(targetId);

                if (targetSection) {
                    // Update active link
                    document.querySelectorAll('.admin-sidebar a').forEach(a => a.classList.remove('active'));
                    this.classList.add('active');

                    // Scroll to section
                    targetSection.scrollIntoView({ behavior: 'smooth' });
                }
            });
        });
